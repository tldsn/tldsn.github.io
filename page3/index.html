SUM(if(status=0,1,0)) success,
SUM(if(status!=0,1,0)) fail,
truncate(SUM(if(status=0,1,0))/count(*)*100,2) successrate,  
max(a.add_time) lastaddtime

where a.add_date  = DATE_FORMAT(now(),'%Y-%m-%d')
